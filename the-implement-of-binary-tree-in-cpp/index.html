void CreateBiTree(BiTree *T)
{

scanf(&quot;%c&quot;, &amp;ch);

    (*T) = NULL;

// 初始条件：二叉树存在
// 操作结果：返回二叉树深度
// 递归实现
int BiTreeDepth(BiTree T)
{
if (!T)
return 0;

int i, j;

if (T-&gt;lchild)
    i = BiTreeDepth(T-&gt;lchild); // 递归求出左子树高度
else
    i = 0;

if (T-&gt;rchild)
    j = BiTreeDepth(T-&gt;rchild); // 递归求出右子树高度
else
    j = 0;

return i &gt; j ? i+1 : j+1;

}

// 求二叉树高度的非递归形式
int BiTreeDepthNonRecursion(BiTree T)
{
if (!T)
return 0;

queue&lt;BiTNode *&gt; Q; // 借助队列实现层次遍历，从而求出高度
BiTNode *p; // 记录队列头部
BiTNode *back; // 记录队列尾部指针
int level = 0; // 队列高度

Q.push(T);
back = Q.back();
while (!Q.empty())
{
    p = Q.front();
    Q.pop();

    if (p-&gt;lchild)
        Q.push(p-&gt;lchild);
    if (p-&gt;rchild)
        Q.push(p-&gt;rchild);

    if (p == back)
    {
        level++;
        if (!Q.empty()) // 防止最后Q为空时出错
            back = Q.back();
    }
}

return level;

}

// 求二叉树高度的非递归形式
int BiTreeDepthNonRecursion2(BiTree T)
{
BiTNode Q[MAXSIZE]; // 借助队列实现层次遍历，从而求出高度。此时用数组实现队列
int level = 0; // 二叉树高度
int last = 0; // 每层次最后一个结点
int front = -1; // 队列头指针
int rear = -1; // 队列尾指针
BiTNode p;

Q[++rear] = T;
last = rear;
while (front &lt; rear) // 队列不空
{
    p = Q[++front];

    if (p-&gt;lchild)
        Q[++rear] = p-&gt;lchild;
    if (p-&gt;rchild)
        Q[++rear] = p-&gt;rchild;

    if (front == last)
    {
        level++;
        if (front &lt; rear)
            last = rear;
    }
}

return level;

}

// 前序遍历非递归形式
void PreOrderNonRecursion(BiTree T)
{
if (!T)
return;

stack&lt;BiTNode *&gt; S; // 借助栈实现非递归遍历
BiTNode *p;

p = T;
while (p || !S.empty())
{
    while (p)
    {
        S.push(p);
        visit(p); // 在每次入栈时进行访问
        p = p-&gt;lchild;
    }
    if (!S.empty())
    {
        p = S.top(); 
        S.pop();
        p = p-&gt;rchild;
    }
}

}

// 中序遍历二叉树非递归形式
void InOrderNonRecursion(BiTree T)
{
if (!T)
return;

stack&lt;BiTNode *&gt; S;
BiTNode *p;

p = T;
while (p || !S.empty())
{
    while (p)
    {
        S.push(p);
        p = p-&gt;lchild;
    }
    if (!S.empty())
    {
        p = S.top();
        S.pop();
        visit(p); // 每次从栈中弹出的时候访问结点
        p = p-&gt;rchild;
    }
}

}

// 初始条件：二叉树存在
// 操作结果：后序递归遍历二叉树
void PostOrderTraverse(BiTree T)
{
if (!T)
return;

PostOrderTraverse(T-&gt;lchild);
PostOrderTraverse(T-&gt;rchild);
visit(T);

}

// 后序遍历非递归形式
// 难点是分清栈中弹出根结点时，是从左子树弹出还是右子树弹出。所以使用辅助指针r。
void PostOrderNonRecurion(BiTree T)
{
if (!T)
return;

stack&lt;BiTNode *&gt; S;
BiTNode *p;
BiTNode *r; // 用指针r记录最近访问的结点

p = T;
r = NULL;
while (p || !S.empty())
{
    while (p)
    {
        S.push(p);
        p = p-&gt;lchild;
    }
    if (!S.empty())
    {
        p = S.top();
        if (p-&gt;rchild &amp;&amp; p-&gt;rchild != r) // 如果右子树存在，且未访问过
        {
            p = p-&gt;rchild;
            S.push(p);
            p = p-&gt;lchild;
        }
        else
        {
            S.pop();
            visit(p);
            r = p;
            p = NULL;
        }
    }
}

}

// 初始条件：二叉树存在
// 操作结果：层次遍历二叉树
void LevelOrderTraverse(BiTree T)
{
if (!T)
return;

queue&lt;BiTNode *&gt; Q; // 借助队列实现层次遍历
BiTNode *p;

Q.push(T);
while (!Q.empty())
{
    p = Q.front();
    Q.pop();
    visit(p); // 每次弹出的时候访问结点

    if (p-&gt;lchild)
        Q.push(p-&gt;lchild);
    if (p-&gt;rchild)
        Q.push(p-&gt;rchild);
}

}

// 初始条件：二叉树存在
// 操作结果：访问每层二叉树最右边的结点
void PrintLastInEachLevel(BiTree T)
{
if (!T) // 如果树为空，则返回
return;

queue&lt;BiTNode *&gt; Q;
BiTNode *p;
BiTNode *back; // 指向每层最后一个结点的指针

Q.push(T);
back = Q.back();
while (!Q.empty())
{
    p = Q.front();
    Q.pop();

    if (p-&gt;lchild)
        Q.push(p-&gt;lchild);
    if (p-&gt;rchild)
        Q.push(p-&gt;rchild);

    if (p == back)
    {
        visit(p);
        if (!Q.empty())
            back = Q.back();
    }
}

}

// 初始条件：二叉树存在
// 操作结果：分行输出每层二叉树的结点
void PrintLevelOrderByLevel(BiTree T)
{
if (!T)
return;

queue&lt;BiTNode *&gt; Q;

BiTNode *back; // 记录每层最右边的结点，从而实现分行输出

Q.push(T);
back = Q.back();
while (!Q.empty())
{
    p = Q.front();
    Q.pop();
    visit(p);

    if (p-&gt;lchild)
        Q.push(p-&gt;lchild);
    if (p-&gt;rchild)
        Q.push(p-&gt;rchild);

    if (p == back)
    {
        putchar('\n');
        if (!Q.empty())
            back = Q.back();
    }
}
